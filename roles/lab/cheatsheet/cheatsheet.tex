\section*{Mamba (Python environments) \hfill\raisebox{-0.2\height}{\includegraphics[height=1.5em]{figure/python.png}}\hspace{0.25cm}\raisebox{-0.2\height}{\includegraphics[height=1.5em]{figure/mamba.png}}\hspace{0.25cm}\raisebox{-0.2\height}{\includegraphics[height=1.5em]{figure/anaconda.png}}}
\label{sec:orgac57fea}
\textbf{What is it?} Mamba is a package manager. It is a drop-in replacement for Conda, but rewritten in C++ to be drastically faster. It isolates your Python versions and libraries so different projects don't conflict.
\vspace{1em}\noindent\rule{\linewidth}{0.4pt}\vspace{-1em}
\subsection*{Create a new environment}
\label{sec:org671129c}
Specify the Python version and any starting packages.
\begin{lstlisting}[language=sh,numbers=none]
mamba create -n <environment_name> python=3.10 pytorch -c pytorch
\end{lstlisting}
\subsection*{List all environments}
\label{sec:org271f041}
See all available environments on the server.
\begin{lstlisting}[language=sh,numbers=none]
mamba env list
\end{lstlisting}
\subsection*{Activate the environment}
\label{sec:org4493de4}
Activate the environment every time you log in before running your code.
\begin{lstlisting}[language=sh,numbers=none]
mamba activate <environment_name>
\end{lstlisting}
\subsection*{Install additional packages}
\label{sec:org4adbdea}
Package installation can be done with \texttt{mamba} or \texttt{pip} (after environment activation).
\begin{lstlisting}[language=sh,numbers=none]
mamba install pandas  # mixing mamba and pip can create dependencies issues
pip install scipy     # if you need to use both, prioritize for mamba over pip
\end{lstlisting}
\subsection*{Deactivate the environment}
\label{sec:orgc17fe3d}
Return to the default system environment.
\begin{lstlisting}[language=sh,numbers=none]
mamba deactivate
\end{lstlisting}
\subsection*{Delete an environment}
\label{sec:orgae7acf1}
Clean up when your experiment is finished.
\begin{lstlisting}[language=sh,numbers=none]
mamba env remove -n <environment_name>
\end{lstlisting}
\vspace{0em}\noindent\rule{\linewidth}{0.4pt}\vspace{0.25em}
\subsection*{Example: train an AI model on MNIST dataset}
\label{sec:orgaecc1e1}
\begin{lstlisting}[language=sh,numbers=none,basicstyle=\fontsize{8pt}{8.5pt}\ttfamily]
mamba create -n mnist_env python=3.10 -y
mamba activate mnist_env
pip install torch torchvision --upgrade --index-url https://download.pytorch.org/whl/cu128
wget -P ~/mnist_test https://raw.githubusercontent.com/pytorch/examples/main/mnist/main.py
cd mnist_test
python main.py --epochs 1
\end{lstlisting}

\section*{Podman (containers) \hfill\raisebox{-0.2\height}{\includegraphics[height=1.5em]{figure/podman.png}}\hspace{0.25cm}\raisebox{-0.2\height}{\includegraphics[height=1.5em]{figure/docker.png}}}
\label{sec:org78d9845}
\textbf{What is it?} Podman is a container engine. It is functionally identical to Docker (you can even use the alias "docker" to run the commands), but it is "daemonless" and runs securely without needing root privileges.
\vspace{0em}\noindent\rule{\linewidth}{0.4pt}\vspace{0em}
\subsection*{Run a container interactively}
\label{sec:org6131432}
Download an image, mount volumes, and run a command inside the container.
\begin{lstlisting}[language=sh,numbers=none]
podman run --rm -it --device nvidia.com/gpu=all \
       --volume /dataset/mnist:/dataset         \
       --volume $(pwd):/workspace               \
       pytorch/pytorch:latest                   \
       bash
\end{lstlisting}
\subsection*{List running containers and downloaded images}
\label{sec:orgaa3ba21}
\begin{lstlisting}[language=sh,numbers=none]
podman ps             # check what is currently running and get the container IDs
podman image ls       # check what container images are currently stored locally
\end{lstlisting}
\subsection*{Enter inside a running container}
\label{sec:orgfc598da}
\begin{lstlisting}[language=sh,numbers=none]
podman exec -it <container_id> bash
\end{lstlisting}
\subsection*{Stop a container}
\label{sec:orgdd042b9}
\begin{lstlisting}[language=sh,numbers=none]
podman stop <container_id>
\end{lstlisting}
\subsection*{Remove a stopped container}
\label{sec:org77bd61d}
\begin{lstlisting}[language=sh,numbers=none]
podman rm <container_id>
\end{lstlisting}
\vspace{0em}\noindent\rule{\linewidth}{0.4pt}\vspace{0.25em}
\subsection*{Example: run a Jupyter Notebook (with GPU)}
\label{sec:org52e6a8b}
\vspace{0.5em}
Run a Jupyter container, map port \texttt{<xxxx>}, and keep it running in the background.
\newline
If it asks for a token, run \texttt{podman logs <container\_id>} to find the login URL.
\begin{lstlisting}[language=sh,numbers=none]
podman run -d --device nvidia.com/gpu=all -p <xxxx>:8888 jupyter/scipy-notebook
\end{lstlisting}
Run this on your local machine to map the server's Jupyter port to your own computer.
\begin{lstlisting}[language=sh,numbers=none]
ssh -L <xxxx>:localhost:<xxxx> -Nf username@server_ip
\end{lstlisting}
Then access \texttt{https://localhost:<xxxx>} on a web browser.
